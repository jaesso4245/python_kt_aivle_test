def solution(a,b):
    if b>a:
        a,b = b,a
    sum = 0
    for i in range(b,a+1):
        count = 0
        for j in range(1,i+1):
            if i%j == 0:
                count +=1
        if count%2 == 0:
            sum += i
        else:
            sum -= i
    
    return sum


x = int(input('정수x는?:'))
y = int(input('정수y는?:'))

print(f'{solution(x,y)}')
